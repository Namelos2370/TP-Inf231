#include <stdio.h>

int mult_par_incrementation(int a, int b) {
    int resultat = 0;

    for (int i = 0; i < b; i++) {     
        for (int j = 0; j < a; j++) {   
            resultat++;                  
        }
    }

    return resultat;
}

int main() {
    int a, b;

    printf("Entrez deux entiers positifs a et b : ");
    scanf("%d %d", &a, &b);

    int produit = mult_par_incrementation(a, b);

    printf("%d Ã— %d = %d\n", a, b, produit);

    return 0;
}
