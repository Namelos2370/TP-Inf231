#include <stdio.h>

int estTrie(int tab[], int n) {
    for (int i = 0; i < n - 1; i++) {
        if (tab[i] > tab[i + 1]) {
            return 0; 
        }
    }
    return 1; 
}

int main() {
    int n;
    printf("Entrez la taille du tableau : ");
    scanf("%d", &n);

    int tab[n];
    printf("Entrez les %d elements :\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &tab[i]);
    }

    if (estTrie(tab, n)) {
        printf("Le tableau est trie en ordre croissant.\n");
    } else {
        printf("Le tableau n'est pas trie.\n");
    }

    return 0;
}
