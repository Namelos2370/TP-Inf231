#include <stdio.h>

void trier(int tab[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min = i;
        for (int j = i + 1; j < n; j++) {
            if (tab[j] < tab[min]) {
                min = j;
            }
        }
        int tmp = tab[i];
        tab[i] = tab[min];
        tab[min] = tmp;
    }
}

double mediane(int tab[], int n) {
    trier(tab, n);
    if (n % 2 == 1) {
        return tab[n / 2];
    } else {
        return (tab[n/2 - 1] + tab[n/2]) / 2.0;
    }
}

int main() {
    int n;
    printf("Entrez la taille du tableau : ");
    scanf("%d", &n);

    int tab[n];
    printf("Entrez les %d elements :\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &tab[i]);
    }

    double med = mediane(tab, n);
    printf("La mediane du tableau est : %.2f\n", med);

    return 0;
}
