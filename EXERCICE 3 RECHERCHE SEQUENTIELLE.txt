#include <stdio.h>

int recherche_sequentielle(int tab[], int n, int cle) {
    for (int i = 0; i < n; i++) {
        if (tab[i] == cle) {
            return i;
        }
    }
    return -1;
}

int main() {
    int n, cle;

    printf("Entrez la taille du tableau : ");
    scanf("%d", &n);

    int tab[n];
    printf("Entrez les %d elements du tableau :\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &tab[i]);
    }

    printf("Entrez la valeur a rechercher : ");
    scanf("%d", &cle);

    int pos = recherche_sequentielle(tab, n, cle);

    if (pos != -1) {
        printf("Valeur %d trouvee a la position %d (indice %d).\n", cle, pos + 1, pos);
    } else {
        printf("Valeur %d non trouvee dans le tableau.\n", cle);
    }

    return 0;
}
